 29: /**
 30:  * Advanced generic Object class.
 31:  * Setups up default behaviour for various magic methods
 32:  *
 33:  * @author Samy Naamani <samy@namani.net>
 34:  * @license https://github.com/sntools/core/blob/master/LICENSE MIT
 35:  */
 36: class Object {
 37: 
 38:     /**
 39:      * Property getters handler.
 40:      * This method is called automatically when trying to read a property.
 41:      * The default behaviour set here is to always throw a PropertyException.
 42:      * Subclasses will want to override this method to add behaviour for their
 43:      * readable properties (example : with switch)
 44:      * @example examples/Person.php
 45:      * @param string $name Property name
 46:      * @return mixed Property value
 47:      * @throws PropertyException
 48:      */
 49:     public function __get($name) {
 50:         throw new PropertyException(sprintf('%s::$%s : invalid property.', get_called_class(), $name));
 51:     }
 52: 
 53:     /**
 54:      * Property setters handler.
 55:      * @example examples/Person.php
 56:      * @param string $name Property name
 57:      * @param mixed $value Property value
 58:      * @throws PropertyException
 59:      */
 60:     public function __set($name, $value) {
 61:         throw new PropertyException(sprintf('%s::$%s : attempt to write invalid property, using value %s.', get_called_class(), $name, print_r($value, true)));
 62:     }
 63: 
 64:     /**
 65:      * Property existance checking handler.
 66:      * @param string $name Property name
 67:      * @return boolean Is the propety set ?
 68:      */
 69:     public function __isset($name) {
 70:         try {
 71:             $ex = $this->$name; // If not set, will throw PropertyException
 72:             // stored in $ex just to trick IDE into thinking $ex is used
 73:             return true;
 74:         } catch (PropertyException $ex) {
 75:             return false;
 76:         }
 77:     }
 78: 
 79:     /**
 80:      * Property unsetter handler.
 81:      * @param string $name Property name
 82:      * @throws PropertyException
 83:      */
 84:     public function __unset($name) {
 85:         throw new PropertyException(sprintf('%s::$%s : invalid attempt to delete property.', get_called_class(), $name));
 86:     }
 87: 
 88:     /**
 89:      * Object to string conversion handler.
 90:      * @return string
 91:      */
 92:     public function __toString() {
 93:         return get_called_class();
 94:     }
 95: 
 96:     /**
 97:      * Object constructor.
 98:      */
 99:     public function __construct() {
100:         static::__constructStatic();
101:     }
102: 
103:     /**
104:      * Static constructor. 
105:      * The default static constructor does nothing besides checking if it has already been loaded.
106:      * Overrides to add new responsabilities that are to be loaded the first time an instance is created.
107:      * @example examples/DAO.php
108:      * @staticvar boolean $init Has been initialized.
109:      * @return boolean Has been initialized
110:      */
111:     protected static function __constructStatic() {
112:         static $init = false;
113:         if ($init)
114:             return true;
115:         $init = true;
116:         return false;
117:     }
118: 
119: }
