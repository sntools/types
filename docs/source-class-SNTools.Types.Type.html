 27: namespace SNTools\Types;
 28: use SNTools\Object;
 29: use SNTools\Types\Autoboxing\Memory;

 31: /**
 32:  * Superclass for all autoboxed types.
 33:  * This class fills the role of the AutoBoxedObject class, according to the Autoboxing technique from Arthur Graniszewski
 34:  *
 35:  * @link http://www.phpclasses.org/package/6570-PHP-Wrap-string-and-integer-values-in-objects.html Arthur Graniszewski's Autoboxing classes
 36:  * @author Samy NAAMANI <samy@namani.net>
 37:  * @license https://github.com/sntools/types/blob/master/LICENSE MIT
 38:  * @property boolean $nullable Is the element nullable or not ?
 39:  */
 40: abstract class Type extends Object {
 41:     /**
 42:      * Real value
 43:      * @var mixed
 44:      */
 45:     protected $value;
 46:     /**
 47:      * Memory variable
 48:      * @var int
 49:      */
 50:     private $memoryId;
 51: 
 52:     /**
 53:      * Is the element nullable or not ?
 54:      * @var boolean
 55:      * @see Type::$nullable
 56:      */
 57:     private $_nullable = false;
 58:     
 59:     public function __get($name) {
 60:         switch($name) {
 61:             case 'nullable':
 62:                 return $this->_nullable;
 63:             default:
 64:                 return parent::__get($name);
 65:         }
 66:     }
 67:     
 68:     public function __set($name, $value) {
 69:         switch($name) {
 70:             case 'nullable':
 71:                 $this->_nullable = (bool)$value;
 72:                 break;
 73:             default:
 74:                 parent::__set($name, $value);
 75:         }
 76:     }
 77:     
 78:     /**
 79:      * Create a new autoboxed variable
 80:      * @param &mixed $var Reference to the variable to create
 81:      * @param mixed|null $value If not null, value to use to override referenced variable
 82:      * @param boolean $override If neither $var nor $value, allow override of $var or not
 83:      * @throws TypeMismatchException
 84:      * @throws InvalidValueException
 85:      * @throws OverrideException
 86:      */
 87:     final public static function create(&$var, $value = null, $override = false) {
 88:         if(!is_null($var)) {
 89:             if(is_null($value)) $value = $var;
 90:             elseif(!$override) throw new OverrideException;
 91:         }
 92:         
 93:         if($value instanceof static) {
 94:             $var = clone $value;
 95:         } else {
 96:             $var = new static($value);
 97:         }
 98:         $var->memoryId = Memory::alloc($var);
 99:     }
100:     
101:     /**
102:      * Private constructor. Use Type::create() instead.
103:      * @param mixed|null $value
104:      * @throws TypeMismatchException
105:      * @throws InvalidValueException
106:      */
107:     private function __construct($value = null) {
108:         parent::__construct();
109:         $this->setValue($value);
110:     }
111: 
112:     /**
113:      * Destructor. Frees variable from memory, OR creates a new variable from previous one
114:      * @ignore
115:      */
116:     final public function __destruct() {
117:         if(!is_null($this->memoryId)) {
118:             $pointer =& Memory::get($this->memoryId);
119:             $value = $pointer;
120:             if($value !== $this and !is_null($value)) {
121:                 $pointer = null;
122:                 static::create($pointer, $value);
123:             }
124:             Memory::free($this->memoryId);
125:         }
126:     }
127:     
128:     /**
129:      * Sets value into variable
130:      * @param mixed $value Value to use
131:      * @throws TypeMismatchException
132:      * @throws InvalidValueException
133:      */
134:     final protected function setValue($value) {
135:         switch(gettype($value)) {
136:             case 'boolean':
137:                 $ok = $this->fromBool($value);
138:                 break;
139:             case 'integer':
140:                 $ok = $this->fromInt($value);
141:                 break;
142:             case 'double':
143:             case 'string':
144:             case 'array':
145:             case 'object':
146:             case 'resource':
147:                 $method = 'from' . ucfirst(gettype($value));
148:                 $ok = $this->$method($value);
149:                 break;
150:             case 'NULL':
151:                 if($this->nullable) $this->value = null;
152:                 else $this->clear();
153:                 $ok = true;
154:                 break;
155:             default:
156:                 throw new TypeMismatchException(sprintf('Unexpected type %s for %s', gettype($value), get_called_class ()));
157:         }
158:         if(!$ok) throw new InvalidValueException(sprintf('Unexpected value %s for %s', $value, get_called_class ()));
159:     }
160:     /**
161:      * Checks variable creation from boolean
162:      * @param boolean $value
163:      * @return boolean
164:      */
165:     protected function fromBool($value) {
166:         return false and $value; // IDE anti-warning trick
167:     }
168:     /**
169:      * Check variable creation from integer
170:      * @param int $value
171:      * @return boolean
172:      */
173:     protected function fromInt($value) {
174:         return false and $value; // IDE anti-warning trick
175:     }
176:     /**
177:      * Check variable creation from double
178:      * @param double $value
179:      * @return boolean
180:      */
181:     protected function fromDouble($value) {
182:         return false and $value; // IDE anti-warning trick
183:     }
184:     /**
185:      * Check variable creation from string
186:      * @param string $value
187:      * @return boolean
188:      */
189:     protected function fromString($value) {
190:         return false and $value; // IDE anti-warning trick
191:     }
192:     /**
193:      * Check variable creation from array
194:      * @param array $value
195:      * @return boolean
196:      */
197:     protected function fromArray(array $value) {
198:         return false and $value; // IDE anti-warning trick
199:     }
200:     /**
201:      * Check variable creation from object
202:      * @param object $value
203:      * @return boolean
204:      */
205:     protected function fromObject($value) {
206:         if($value instanceof self) {
207:             $this->setValue($value->value);
208:             return true;
209:         } else return false;
210:     }
211:     /**
212:      * Check variable creation from resource
213:      * @param resource $value
214:      * @return boolean
215:      */
216:     protected function fromResource($value) {
217:         return false and $value; // IDE anti-warning trick
218:     }
219:     /**
220:      * Get inner value
221:      * @return mixed
222:      */
223:     final public function getValue() {
224:         return $this->value;
225:     }
226: 
227:     /**
228:      * clears inner value, setting it to a default value
229:      */
230:     abstract protected function clear();
231:     
232:     /**
233:      * Conversion to boolean
234:      * @return Bool
235:      */
236:     final public function toBool() {
237:         $bool = null;
238:         Bool::create($bool, $this);
239:         return $bool;
240:     }
241: 
242:     /**
243:      * (bool) operator overriding
244:      * @return boolean
245:      * @todo Query : return a native boolean or a Bool object ?
246:      */
247:     public function __bool() {
248:         return $this->toBool()->getValue();
249:     }
250: 
251:     /**
252:      * ! operator overriding
253:      * @return boolean
254:      * @todo Query : return a native boolean or a Bool object ?
255:      */
256:     public function __bool_not() {
257:         return !$this->__bool();
258:     }
259: }
